# if a specific template is used
# Arguments:
# a = mod code
# b = a+1
brs_has_template = {
	custom_trigger_tooltip = {
		tooltip = brs_has_template_$a$_tt
		if = {
			limit = { has_country_flag = brs_template_used }
			brs_check_template = { a=$a$ b=$b$ }
		}
		else = { always = no }
	}
}
brs_not_has_template = {
	custom_trigger_tooltip = {
		tooltip = brs_has_template_$a$_tt
		if = {
			limit = { has_country_flag = brs_template_used }
			NOT = { brs_check_template = { a=$a$ b=$b$ } }
		}
		else = { always = yes }
	}
}
brs_has_template_hidden = {
	hidden_trigger = { brs_has_template = { a=$a$ b=$b$ } }
}
brs_not_has_template_hidden = {
	hidden_trigger = { brs_not_has_template = { a=$a$ b=$b$ } }
}



# for menus, slightly more resources used but also checks if user has selected adything before
# Arguments:
# a = <0|1|2|3|4|...|14|15>
# b = <a+1>
brs_check_target_setup = {
	custom_trigger_tooltip = {
		tooltip = brs_has_target_$a$_tt
		root = {
			if = {
				limit = { NOT = { has_country_flag = brs_setup_done } }
				always = no
			}
			else = {
				check_variable = {
					which = brs_target_var
					value = $a$
				}
				NOT = {
					check_variable = {
						which = brs_target_var
						value = $b$
					}
				}
			}
		}
	}
}

brs_check_var = {
	hidden_trigger = {
		root = {
			[[a]
			check_variable = {
				which = $which$
				value = $a$
			}
			]
			[[b]
			NOT = {
				check_variable = {
					which = $which$
					value = $b$
				}
			}
			]
		}
	}
}

# Arguments:
# a = <0|1|2|3|4|...|14|15|...>
# b = <a+1>
brs_check_target = {
	custom_trigger_tooltip = {
		tooltip = brs_has_target_$a$_tt
		brs_check_var = { which = brs_target_var a = $a$ [[b] b = $b$ ] }
	}
}

# requires using brs_load_maxskill_locally
# maxskill is exactly x ---> a=x b=(x+1)
# maxskill is at least x --> a=x
# maxskill is at most x ---> b=(x+1)
brs_check_maxskill = {
	hidden_effect = { #custom_trigger_tooltip = {
		#tooltip = brs_maxskill_exact_$a$
		if = { # show all options
			limit = { has_country_flag = brs_showalloptions }
			always = yes
		}
		else = {
			1 = {
				brs_check_var = { which = brs_maxskill [[a] a = $a$ ] [[b] b = $b$ ] }
			}
		}
	}
}

# why are there two?
# When the game parses an empty argument (no argument) in a trigger, if it is the first character of a value, the game will throw an error and ignore the command!

# Arguments:
# value = <ADM|DIP|MIL>
brs_check_ruler = {
	variable_arithmetic_trigger = {
		export_to_variable = {
			variable_name = ruler_skill
			value = $value$
		}
		check_variable = {
			which = brs_target_var
			which = ruler_skill
		}
	}
}
