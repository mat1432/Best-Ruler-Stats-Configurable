country_event = {
	id = brs.1
	title = "brs_1_t"
	desc = "brs_1_d"
	picture = DIPLOMACY_eventPicture
	fire_only_once = no
	hidden = yes

	trigger = {
		ai = no
		has_country_flag = brs_setup_done

		OR = {
			brs_check_ruler = { value = ADM }
			brs_check_ruler = { value = DIP }
			brs_check_ruler = { value = MIL }
			if = {
				limit = { has_heir = yes }
				brs_check_heir = { value = adm }
				brs_check_heir = { value = dip }
				brs_check_heir = { value = mil }
			}
			if = {
				limit = { has_consort = yes }
				brs_check_consort = { value = adm }
				brs_check_consort = { value = dip }
				brs_check_consort = { value = mil }
			}
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	after = {
		while = {
			# only 1 check is required
			limit = {
				variable_arithmetic_trigger = {
					export_to_variable = {
						variable_name = ruler_skill
						value = ADM
					}
					change_variable = {
						which = ruler_skill
						value = 1
					}
					check_variable = {
						which = brs_target_var
						which = ruler_skill
					}
				}
			}
			# player chose wrong, chose 1 smaller
			if = {
				limit = { brs_has_any_template = yes }
				brs_clr_template = yes
			}
			subtract_variable = {
				which = brs_target_var
				value = 1
			}
		}
	}

	option = {
		trigger = { brs_minimum_disabled = yes }
		name = "brs_1_normal"
		brs_update_ruler_add = { value = ADM }
		brs_update_ruler_add = { value = DIP }
		brs_update_ruler_add = { value = MIL }
		if = {
			limit = { has_heir = yes }
			brs_update_heir_add = { value = adm }
			brs_update_heir_add = { value = dip }
			brs_update_heir_add = { value = mil }
		}
		if = {
			limit = { has_consort = yes }
			if = {
				limit = { brs_optionsxt_enabled = yes }
				brs_updatext_consort_add = { value = adm }
				brs_updatext_consort_add = { value = dip }
				brs_updatext_consort_add = { value = mil }
			}
			else = {
				brs_update_consort_add = { value = adm }
				brs_update_consort_add = { value = dip }
				brs_update_consort_add = { value = mil }
			}
		}
	}

	option = {
		trigger = { brs_minimum_enabled = yes }
		name = "brs_1_minimum"
		brs_update_ruler = { value = ADM }
		brs_update_ruler = { value = DIP }
		brs_update_ruler = { value = MIL }
		if = {
			limit = { has_heir = yes }
			brs_update_heir = { value = adm }
			brs_update_heir = { value = dip }
			brs_update_heir = { value = mil }
		}
		if = {
			limit = { has_consort = yes }
			if = {
				limit = { brs_optionsxt_enabled = yes }
				brs_updatext_consort = { value = adm }
				brs_updatext_consort = { value = dip }
				brs_updatext_consort = { value = mil }
			}
			else = {
				brs_update_consort = { value = adm }
				brs_update_consort = { value = dip }
				brs_update_consort = { value = mil }
			}
		}
	}
}
